#include <stdio.h>
#include <string.h>
#include <stdlib.h>

int main() {
    char s[1000];
    int freq[10] = {0};
    int i;

    scanf("%[^\n]", s);

    for (i = 0; s[i] != '\0'; i++) {
        if (s[i] >= '0' && s[i] <= '9') {
            freq[s[i] - '0']++;
        }
    }

    for (i = 0; i < 10; i++) {
        printf("%d ", freq[i]);
    }

    return 0;
}
