#include <stdio.h>

int main() {
    int n;
    scanf("%d", &n);

    int size = 2 * n - 1; // total rows and columns

    for (int i = 0; i < size; i++) {
        for (int j = 0; j < size; j++) {
            // Find the minimum distance to any edge
            int min = i < j ? i : j;
            if ((size - 1 - i) < min) min = size - 1 - i;
            if ((size - 1 - j) < min) min = size - 1 - j;

            printf("%d ", n - min);
        }
        printf("\n");
    }

    return 0;
}
