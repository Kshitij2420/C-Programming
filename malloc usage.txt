#include <stdio.h>
#include <stdlib.h>

int main() {
    int n, i, sum = 0;
    int *arr;

    scanf("%d", &n);                         // read size
    arr = (int*) malloc(n * sizeof(int));    // allocate memory

    for (i = 0; i < n; i++) {
        scanf("%d", &arr[i]);                // read elements
        sum += arr[i];                       // sum as we go
    }

    printf("%d", sum);                       // print result

    free(arr);                               // free memory
    return 0;
}
