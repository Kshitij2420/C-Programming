#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int next_permutation(int n, char **s)
{
	 int i = n - 2;
    // Step 1: find first i such that s[i] < s[i+1]
    while (i >= 0 && strcmp(s[i], s[i + 1]) >= 0)
        i--;
    
    if (i < 0)
        return 0; // no next permutation

    // Step 2: find j such that s[j] > s[i]
    int j = n - 1;
    while (strcmp(s[j], s[i]) <= 0)
        j--;

    // Step 3: swap s[i] and s[j]
    char *temp = s[i];
    s[i] = s[j];
    s[j] = temp;

    // Step 4: reverse s[i+1 ... n-1]
    int start = i + 1, end = n - 1;
    while (start < end)
    {
        temp = s[start];
        s[start] = s[end];
        s[end] = temp;
        start++;
        end--;
    }

    return 1;
}
